16  /**
17   * This Mojo creates a module-tree containing all files to be included in the module. It does so by copying all files
18   * that dont match the excludes2 to the distdir.
19   *
20   * @goal    createtree
21   * @phase   package
22   * @author  Marc Kannegiesser, kannegiesser@synyx.de
23   */
24  public class CreateModuleTreeMojo extends AbstractOpenCmsMojo {
25  
26      /**
27       * If you provide targetLibPath the file found under jarFile is copied to it. This defaults to
28       * ${project.build.directory}/${project.build.finalName}.jar
29       *
30       * @parameter  expression="${project.build.directory}/${project.build.finalName}.jar"
31       */
32      private String jarFile = null;
33  
34      /**
35       * Directories in sourceDir to exclude (ant-style excludes). You may provide excludes in ant-style patterns. Files
36       * matching that pattern will not be included to the module (basically they will not be copied to distDir).
37       *
38       * @parameter
39       */
40      private String excludes = null;
41  
42      /**
43       * Defines a set of resources that are included in the module. Currently only directory and targetPath are
44       * evaluated. The targetPath configured is relative to the distFolder. A possible configuration might look like
45       * this:
46       *
47       * <pre>
48      <srcResources>
49      <resource>
50      <directory>${project.basedir}/web/</directory>
51      <targetPath></targetPath>
52      </resource>
53      <resource>
54      <directory>${project.basedir}/../web-module/src/main/webapp/WEB-INF/</directory>
55      <targetPath>system/modules/my.module/WEB-INF/</targetPath>
56      </resource>
57      </srcResources>
58       * </pre>
59       *
60       * @parameter
61       * @required
62       */
63      private Resource[] srcResources;
64  
65      /**
66       * Executes this task and creates the module-tree within the distDir.
67       *
68       * @throws  org.apache.maven.plugin.MojoExecutionException
69       * @throws  org.apache.maven.plugin.MojoFailureException
70       */
71      public void execute() throws MojoExecutionException, MojoFailureException {
72  
73          File target = new File(getDistfolder());
74  
75          // only create the target directoty when it doesn't exist
76          // no deletion
77          if (!target.exists()) {
78              target.mkdir();
79          } else {
80              getLog().warn("Using existing target directory " + target.getAbsolutePath());
81          }
82  
83          getLog().info("Target is " + target);
84  
85          String excludes2 = getExcludes();
86  
87          if (excludes2 == null) {
88              excludes2 = "**/.svn/,**/.cvs/";
89  
90              getLog().info("Configuration for excludes not set. Defaulting to: \"" + excludes2
91                  + "\". Excludes are ant-fileset-style.");
92          } else {
93              if (excludes2.contains("__acl") || excludes2.contains("__properties")) {
94                  getLog().warn("Your excludes \"" + excludes2
95                      + "\" contains __acl or __properties. You do not want these to be excluded in order to have permissoins and properties work.");
96              }
97          }
98  
99          try {
100             String targetLibPath = getTargetLibPath();
101 
102             if ((null != targetLibPath) && !"".equals(targetLibPath)) {
103                 File targetLibDir = new File(target, targetLibPath);
104 
105                 if (!targetLibDir.exists()) {
106                     targetLibDir.mkdirs();
107                 }
108 
109                 if (!targetLibDir.isDirectory()) {
110                     throw new MojoExecutionException("Could not create targetLibdir: " + targetLibDir.toString());
111                 }
112 
113                 File jar = new File(jarFile);
114                 FileUtils.copyFileToDirectory(jar, targetLibDir);
115             }
116 
117             for (Resource resource : srcResources) {
118                 File source = new File(resource.getDirectory());
119                 File currentTarget = null;
120 
121                 if (resource.getTargetPath() != null) {
122                     currentTarget = new File(target, resource.getTargetPath());
123                 } else {
124                     currentTarget = target;
125                 }
126 
127                 String normalized = source.getAbsolutePath();
128                 int offsetLength = normalized.length();
129 
130                 getLog().debug("Normalzed sourcefolder is " + normalized + " (Offset is: " + offsetLength + ")");
131 
132                 List<String> results = FileUtils.getFileAndDirectoryNames(source, null, excludes2, true, true, true,
133                         true);
134 
135                 for (String fileName : results) {
136                     File file = new File(fileName);
137 
138                     String normalizedParent = new File(file.getParent()).getAbsolutePath();
139 
140                     if (offsetLength > normalizedParent.length()) {
141                         getLog().info("Skipping file (its the sourcefolder or a parent of it): "
142                             + file.getAbsolutePath());
143                         getLog().debug("Parent: " + normalizedParent + " with length " + normalizedParent.length());
144 
145                         continue;
146                     }
147 
148                     String relative = file.getParent();
149                     relative = relative.substring(offsetLength); // cut the target-dir away
150 
151                     getLog().debug("Copy " + file + " to " + currentTarget.getPath() + File.separator + relative);
152 
153                     if (file.isFile()) {
154                         File parent = new File(currentTarget.getPath() + File.separator + relative);
155 
156                         if (!parent.exists()) {
157                             // create dir if it does not exist
158                             parent.mkdirs();
159                         }
160 
161                         FileUtils.copyFileToDirectory(file, parent);
162                     } else if (file.isDirectory()) {
163                         // take care about directories (so that empty dirs get created)
164 
165                         File tocreate = new File(currentTarget.getPath() + File.separator + relative + File.separator
166                                 + file.getName());
167                         getLog().debug("Creating dir" + tocreate.toString());
168                         tocreate.mkdirs();
169                     }
170                 }
171             }
172         } catch (IOException ex) {
173             throw new MojoExecutionException("Could copy file(s) to directory", ex);
174         }
175     }
