14  /**
15   * This MOJO starts a CmsShell (OpenCms) and executes any commands there.
16   * 
17   * There are several ways to give commands to the Shell: 
18   * <pre>
19   * 1st:
20   * -----
21   * Provide a File containing the commands and set configuration-node "commandFile" to that file. 
22   * Alternatively provide the property -D cmsshell.commandfile=/path/to/file
23   * 
24   * In this case all other commands given are ignored (commands and additionalcommand).
25   * 
26   * 2nd:
27   * -----
28   * Enter the whole commands using configuration-node "commands".
29   * Alternatively provide the property -D cmsshell.commands=myCommands
30   * If the configuration-node additionalcommands (or the property cmsshell.additionalcommands) 
31   * is also set these are appended at the end of the "regular" commands.
32   * 
33   * 3rd:
34   * -----
35   * Leave out commandFile and commands which results in defaulting to (delete module, import module). 
36   * Additionalcommands can also be given by configuration-node additionalcommands (or the property 
37   * cmsshell.additionalcommands). These are executed after the import.
38   * </pre>
39   *
40   * @phase   install
41   * @goal    cmsshell
42   * @author  Marc Kannegiesser, kannegiesser@synyx.de
43   * @author  Florian Hopf, hopf@synyx.de
44   */

111     private String addLineBreaks(String input) {
112         String[] parts = input.split(";");
113         StringBuilder output = new StringBuilder();
114         for (String part : parts) {
115             output.append(part);
116             output.append("\n");
117         }
118 
119         return output.toString();
120     }

122     /**
123      * Executes this task. 
124      * This basically creates a CMSShell, writes any commands to a temporary file and
125      * lets the shell execute the commands. 
126      *
127      * @throws  org.apache.maven.plugin.MojoExecutionException
128      * @throws  org.apache.maven.plugin.MojoFailureException
129      */
130     public void execute() throws MojoExecutionException, MojoFailureException {
131 
132         //File modulePackageDir = new File(wepappfolder + File.separator + "packages" + File.separator + "modules");
133 
134         File webinf = new File(webappfolder);
135 
136 
137         if (!webinf.isDirectory()) {
138 
139             throw new MojoExecutionException("OpenCms-Webapp directory does not exist: " + webappfolder);
140         }
141 
142 
143         CmsShell shell = new CmsShell(webappfolder, "opencms", "opencms", "", null);
144 
145 
146         File commandsToExecute = null;
147 
148         boolean deleteCommandsToExecute = true;
149 
150         if (commandFile != null && !"".equals(commandFile)) {
151             commandsToExecute = new File(commandFile);
152             if (!commandsToExecute.exists()) {
153                 throw new MojoExecutionException("commandFile does not exist: " + commandsToExecute.getAbsolutePath());
154             }
155             deleteCommandsToExecute = false;
156 
157         } else { // we need to create a tempfile since opencms wants a FileInputSource...
158 
159             String shellCommand = "";
160 
161             // if configuration has a commands we use it.
162             if (commands != null && !"".equals(commands)) {
163 
164                 shellCommand = addLineBreaks(commands);
165             } else { // otherwise we default to import
166 
167                 File zipFile = new File(getModuleDir(), getName() + "_" + getVersion() + ".zip");
168 
169                 shellCommand = "login \"" + username + "\" \"" + password + "\"\n";
170                 if ( changetositeroot != null) {
171                     shellCommand += "setSiteRoot \"" + changetositeroot + "\"\ngetSiteRoot\n";
172                 }
173                 shellCommand += "deleteModule " + getName() + "\nimportModule \"" + zipFile.getAbsolutePath() + "\"\n";
174             }
175 
176             if (additionalcommands != null && !"".equals(additionalcommands)) {
177                 shellCommand += "\n" + addLineBreaks(additionalcommands);
178             }
179 
180             try {
181                 shellCommand = escapeForCmsShell(shellCommand);
182                 getLog().info("I will execute the following commands:\n====\n" + shellCommand + "\n====\n" );
183                 commandsToExecute = File.createTempFile(this.getClass().getName(), null);
184 
185                 FileWriter writer = new FileWriter(commandsToExecute);
186 
187                 writer.write(shellCommand);
188                 writer.close();
189 
190             } catch (IOException ex) {
191 
192                 throw new MojoExecutionException("Could not create or write temporary file for shellcommand", ex);
193             }
194 
195 
196         }
197 
198         // now read the stream again
199         FileInputStream fis;
200         try {
201 
202             fis = new FileInputStream(commandsToExecute);
203         } catch (FileNotFoundException ex) {
204             if (deleteCommandsToExecute) {
205                 commandsToExecute.delete();
206             }
207             throw new MojoExecutionException("Cannot create FileInputStream" + ex);
208         }
209 
210 
211         // give it to cmsshell
212         try {
213 
214             shell.start(fis);
215 
216         } catch (Exception ex) {
217             if (deleteCommandsToExecute) {
218                 commandsToExecute.delete();
219             }
220             throw new MojoExecutionException("Exceptoin while running commands in CmsShell", ex);
221         }
222 
223 
224         shell.exit();
225 
226         // if we created the file we will delete it afterwards
227         if (deleteCommandsToExecute) {
228             commandsToExecute.delete();
229         }
230 
231     }
