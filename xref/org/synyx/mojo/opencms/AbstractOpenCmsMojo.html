140     private String extractVersionNumber(String originalVersionNumber) {
141         // check if there are non-digit letters at the end
142         if (originalVersionNumber != null && originalVersionNumber.matches(".*\\D+$")) {
143             // remove anything after the last digit
144             originalVersionNumber = originalVersionNumber.replaceAll("\\D+$", "");
145         }
146         return originalVersionNumber;
