182     private void convertParameters(ManifestBuilderTask anttask) {
183 
184         // convert parameters into anttask-parameters
185         if (parameters != null) {
186 
187             for (Object okey : parameters.keySet()) {
188 
189                 String key = (String) okey;
190                 Parameter antparameter = anttask.createParameter();
191 
192 
193                 antparameter.setName(key);
194                 antparameter.setValue(parameters.getProperty(key));
195             }
196         }
197     }

200      * This method creates Parameters the way the ant-task likes them
201      *
202      * @param  anttask  ManifestBuilderTask
203      */
204     private void convertExportPoints(ManifestBuilderTask anttask) {
205 
206         // convert parameters into anttask-parameters
207         if (exportpoints != null) {
208 
209             for (Object okey : exportpoints.keySet()) {
210 
211                 String key = (String) okey;
212                 ExportPoint antparameter = anttask.createExportPoint();
213 
214 
215                 antparameter.setSrc(key);
216                 antparameter.setDst(exportpoints.getProperty(key));
217             }
218         }
219 
220 
221         String targetLibPath = getTargetLibPath();
222 
223 
224         if ((null != targetLibPath) && !"".equals(targetLibPath)) {
225 
226             // eventually add leading slash /
227             if (!targetLibPath.startsWith(File.separator)) {
228 
229                 targetLibPath = File.separator + targetLibPath;
230             }
231 
232 
233             File targetLibDir = new File(new File(getDistfolder()), targetLibPath);
234 
235 
236             if (targetLibDir.exists() && !exportpoints.containsKey(targetLibPath)) {
237 
238                 ExportPoint antparameter = anttask.createExportPoint();
239                 getLog().info("Adding additional exportpoint: " + targetLibPath);
240                 antparameter.setSrc(targetLibPath);
241                 antparameter.setDst("WEB-INF/lib/");
242             }
243         }
244     }
