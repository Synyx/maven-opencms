716 
717             if (new File(propFilePath).exists())
718                 props.load(new FileInputStream(propFilePath));
719 
720

722 
723                 for (Object keyObject : props.keySet()) {

725                     try {

756

769

777     /**
778      * This method adds an accesscontrol tag to the given resource. Therefore
779      * it parses the attributes of the flag, allowed and denied tag to calculate
780      * the integer value.
781      * @param root The root element to add at the accesscontrol tag.
782      * @param propFilePath The path to the property file.
783      * 
784      * @author Rainer Steinegger - Synyx GmbH & Co. KG - rainer.steinegger@synyx.de
785      */
786     protected void addConfiguredAccessesToTree(Element root, String propFilePath) {
787         
788         if (null != propFilePath) {
789             File xml = new File(propFilePath);
790             if ( ! xml.exists()) {
791                 return;
792             }
793             
794             SAXReader reader = new SAXReader();
795             Document doc;
796 
797 
798             try {
799 
800                 doc = reader.read(xml);
801                 // generate a list with the searched tags
802                 ArrayList<A_AttributeConverter> permissions = new ArrayList<A_AttributeConverter>();
803                 permissions.add(new FlagConverter(CmsImportExportManager.N_FLAGS, 
804                         CmsImportExportManager.N_ACCESSCONTROL_ENTRY));
805                 permissions.add(new PermissionConverter(CmsImportExportManager.N_ACCESSCONTROL_ALLOWEDPERMISSIONS, 
806                         CmsImportExportManager.N_ACCESSCONTROL_PERMISSIONSET));
807                 permissions.add(new PermissionConverter(CmsImportExportManager.N_ACCESSCONTROL_DENIEDPERMISSIONS, 
808                         CmsImportExportManager.N_ACCESSCONTROL_PERMISSIONSET));
809                 replaceAttributes(doc.getRootElement(), permissions);
810 
811                 Element elem = doc.getRootElement();
812 
813 
814                 if (null != elem) {
815 
816                     root.add(elem);
817                 } else {
818 
819                     root.addElement("accesscontrol");
820                 }
821             } catch (DocumentException e) {
822                 System.out.println(e.toString());
823                 root.addElement("accesscontrol");
824             } catch (Exception ex) {
825                 System.out.println(ex.toString());
826                 root.addElement("accesscontrol");
827             }
828         }
829     }

831     /**
832      * This method replaces the attributes given at the tag with
833      * an calculated Integer value.
834      * @param rootElement The element to evaluate.
835      * @param converter A list of A_AttributeConverter to convert the attributes of the searched tag.
836      * @throws Exception 
837      * 
838      * @author Rainer Steinegger - Synyx GmbH & Co. KG - rainer.steinegger@synyx.de
839      */
840     private void replaceAttributes(Element rootElement, ArrayList<A_AttributeConverter> converter) throws Exception {        
841         if (rootElement != null) {
842             List<Element> elements = (List<Element>) rootElement.elements();            
843             if (elements != null) {
844                 for (Element currentElement : elements) {                                            
845                     replaceAttributes(currentElement, converter);                                    
846                 }
847             }            
848             
849             for (A_AttributeConverter currentConverter : converter) {
850                 if (currentConverter.isSearchedTag(rootElement)) {                            
851                     currentConverter.convertTag(rootElement);
852                 }
853             }
854         }
855         
856     }
